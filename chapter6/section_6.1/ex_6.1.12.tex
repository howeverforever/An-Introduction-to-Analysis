\begin{Exercise}
\begin{proof}
Let 
$$
b_n 
:= \sum_{k=1}^{n} k a_k 
= \frac{n+1}{n+2}. 
$$
So 
$$
b_{n+1}-b_n
= (n+1)a_{n+1}
= \frac{1}{(n+2)(n+3)}. 
$$
We have 
$$
a_1=\frac{2}{3}\text{ and } a_n=\frac{1}{n(n+1)(n+2)}\mbox{ for }n > 1.
$$
Calculate the summation and we get
$$
\sum_{k=1}^{\infty} a_k 
= a_1+\sum_{k=2}^{\infty} a_k 
= \frac{2}{3}+\frac{1}{12} = \frac{3}{4}.
$$
\end{proof}
\end{Exercise}
